navigator.geolocation.getCurrentPosition(

  function(position) {

    const lat = position.coords.latitude;

    const lon = position.coords.longitude;


    const formData = new FormData();

    formData.append('entry.1077879434', lat); // Replace with your Latitude field ID

    formData.append('entry.1017437908', lon);  // Replace with your Longitude field ID


    fetch('https://docs.google.com/forms/d/e/1FAIpQLSdo_zmFieU_04f7nuP6iN04igkQ7UX91pJr5T-YevJ44Aaj8w/formResponse', {

      method: 'POST',

      body: formData,

      mode: 'no-cors'

    });


    console.log('Location submitted:', lat, lon);

  },

  function(error) {

    console.error('Error getting location:', error);

  },

  { enableHighAccuracy: true }

);

</script>
